\section{Legi de compoziție (interne). Subgrup (18-Oct-1994)}

Fie $(S, \varphi)$, cu $S \neq \emptyset$ și $\varphi: S \times S \to S$.

\begin{example}\leavevmode
\begin{itemize}
    \item $A = \text{mulțime dată}$, $(\mathcal{P}(A), \cup)$ sau $(\mathcal{P}(A), \cap)$.
    \item $(\{f \mid f: [a, b] \to \mathbb{R}\}, +)$.
\end{itemize}
\end{example}

\begin{definition}
Dacă $S$ este o mulțime $S \neq \emptyset$, o aplicație $\varphi: S \times S \to S$ se numește \textbf{lege de compoziție (internă)} pe $S$.
$$ S \times S \to S $$
$$ (x, y) \mapsto \varphi(x, y) \text{ se numește compusul lui } x \text{ cu } y $$
\end{definition}

\textbf{Notații:} $x * y, x + y$ (aditivă), $x \cdot y$ (multiplicativă). \\
Alte notații pentru $\varphi(x, y)$: $x \circ y$, $x \top y$, $x \bot y$, $x \square y$.

\begin{definition}[Proprietăți]
O lege de compoziție $\varphi$ este:
\begin{itemize}
    \item \textbf{asociativă} dacă $\varphi(x, \varphi(y, z)) = \varphi(\varphi(x, y), z)$, $(\forall) x, y, z \in S$.
    \item \textbf{comutativă} dacă $(\forall) x, y \in S \implies \varphi(x, y) = \varphi(y, x)$.
\end{itemize}
\end{definition}

\begin{definition}
Un element $e \in S$ se numește \textbf{element neutru} pentru $\varphi$ dacă:
$$ (\forall) x \in S \text{ avem } \varphi(e, x) = \varphi(x, e) = x $$
\end{definition}

\begin{proposition}
Dacă o lege de compoziție $\varphi$ admite element neutru, atunci el este unic.
\end{proposition}
\begin{proof}
Fie $e$ și $e'$ două elemente neutre.
\begin{itemize}
    \item Dacă privim $e$ ca element neutru: $e' \varphi e = e \varphi e' = e'$.
    \item Dacă privim $e'$ ca element neutru: $e \varphi e' = e' \varphi e = e$.
\end{itemize}
Prin urmare, $e = e'$.
\end{proof}

\begin{example}
$(2\mathbb{N}, \cdot)$ nu are element neutru. (Elementul neutru pentru înmulțire ar fi $1$, dar $1 \notin 2\mathbb{N}$).
\end{example}

\begin{definition}
Fie $(S, \varphi)$ o lege de compoziție cu element neutru $e$. Un element $x \in S$ este \textbf{simetrizabil} în raport cu $\varphi \iff (\exists) x' \in S$ astfel încât:
$$ \varphi(x, x') = \varphi(x', x) = e $$
\end{definition}

\begin{proposition}
1) Dacă $\varphi$ este asociativă și are un element neutru, iar $x$ admite simetric $\implies$ acesta este unic. \\
2) Dacă $x, y \in S$ sunt simetrizabile, atunci $\varphi(x, y)$ este simetrizabil și $(\varphi(x, y))' = \varphi(y', x')$.
\end{proposition}

\begin{proof}
1) Fie $x', x'' \in S$ astfel încât:
\begin{align*}
    x x' &= x' x = e \\
    x x'' &= x'' x = e
\end{align*}
$$ x'' = x'' e = x'' (x x') = (x'' x) x' = e x' = x' $$
\end{proof}

\begin{definition}
Fie $(S, \cdot)$ cu $\cdot$ asociativă, $x_1, \dots, x_n \in S$.
$$ x_1 \cdot x_2 \cdot \dots \cdot x_n = \begin{cases} x_1, & n=1 \\ (x_1 \dots x_{n-1})x_n, & n \ge 2 \end{cases} = \prod_{i=1}^n x_i $$
\end{definition}

\begin{lemma}
Fie $(S, \cdot)$ cu $\cdot$ asociativă. Atunci $(\forall) x_1, \dots, x_n \in S$, $n \in \mathbb{N}^*$, $m \in \mathbb{N}^*$, $1 \le m < n$:
$$ (x_1 \dots x_m)(x_{m+1} \dots x_n) = x_1 \dots x_n $$
\end{lemma}

\begin{proof}
Inducție după $n$. \\
Pentru $n=3$, partițiile posibile sunt: \\
$m=1 \implies x_1(x_2 x_3)$ \\
$m=2 \implies (x_1 x_2)x_3$
\end{proof}

\begin{corollary}
Dacă $(S, \cdot)$ este asociativă, $(\forall) x_1, \dots, x_n \in S$, $n \ge 3$ și $(\forall) m_1, \dots, m_r \in \mathbb{N}^*$ cu $m_1 + \dots + m_r = n$, avem:
$$ (x_1 \dots x_{m_1})(x_{m_1+1} \dots x_{m_1+m_2}) \dots (x_{n-m_r+1} \dots x_n) = x_1 \dots x_n $$
\end{corollary}

\begin{example}
(Generalizarea comutativității). Presupunem că $(S, \cdot)$ este comutativă. Se arată că $(\forall) x_1, \dots, x_n \in S$, $n \ge 2$ și $(\forall) \sigma \in S_n$:
$$ x_{\sigma(1)} x_{\sigma(2)} \dots x_{\sigma(n)} = x_1 x_2 \dots x_n $$
\end{example}

\begin{definition}
Fie $S \neq \emptyset$. O pereche $(S, \varphi)$ unde $\varphi$ este o lege de compoziție pe $S$ se numește \textbf{grupoid}.
\end{definition}

\begin{example}\leavevmode
\begin{itemize}
    \item $(\mathbb{N}, +), (\mathbb{N}, \cdot), (\mathbb{Z}, +)$ etc. sunt grupoizi.
    \item $(\mathcal{M}_n(\mathbb{C}), +), (\mathcal{M}_n(\mathbb{C}), \cdot)$ sunt grupoizi.
\end{itemize}
\end{example}

\begin{definition}
Un grupoid $(S, \varphi)$ se numește \textbf{semigrup} dacă $\varphi$ este asociativă.
\end{definition}

\begin{example}
Toate cele de mai sus.
\end{example}

\begin{definition}
Un semigrup care admite element neutru se numește \textbf{monoid}.
\end{definition}

\begin{example}\leavevmode
\begin{itemize}
    \item $(\mathbb{N}, +) \rightarrow e = 0$; $(\mathbb{N}, \cdot) \rightarrow e = 1$ (la fel pentru $\mathbb{Z}, \mathbb{Q}, \mathbb{R}, \mathbb{C}$).
    \item $(\mathcal{M}_n(\mathbb{C}), +) \rightarrow e = O_n$; $(\mathcal{M}_n(\mathbb{C}), \cdot) \rightarrow e = I_n$.
    \item $(\mathcal{P}(A), \cup) \rightarrow e = \emptyset$; $(\mathcal{P}(A), \cap) \rightarrow e = A$; $(\mathcal{P}(A), \Delta) \rightarrow e = \emptyset$.
    \item $(\mathcal{P}(A), \setminus)$ - nu (nu are element neutru).
    \item $(A^A, \circ)$ unde $A^A = \{f \mid f: A \rightarrow A\}$, elementul neutru este $1_A$.
\end{itemize}
\end{example}

\vspace{1em}
\textbf{Reguli de calcul într-un semigrup}

Fie $(S, \cdot)$ un semigrup și $x \in S$.
\begin{align*}
    x^1 &= x \\
    x^2 &= x \cdot x \\
    &\dots \\
    x^n &= \underbrace{x \cdot x \dots x}_{n \text{ ori}}
\end{align*}
$$ x^n \cdot x^m = x^{n+m} $$
$$ (x^n)^m = x^{n \cdot m} $$
Dacă $S$ este monoid: $x^0 \stackrel{\text{def}}{=} e$ (convenție).

\textit{Forma aditivă:}
\begin{align*}
    1 \cdot x &= x \\
    2x &= x + x \\
    &\dots \\
    nx &= \underbrace{x + x \dots x}_{n \text{ ori}}
\end{align*}
$$ nx + mx = (n+m)x $$
$$ m(nx) = (mn)x $$
Dacă $S$ este monoid: $0 \cdot x = 0$.

\subsection*{Subgrupuri}

\begin{definition}
Fie $(G, \cdot)$ un grup și $H \subseteq G$, $H \neq \emptyset$. Submulțimea $H$ se numește \textbf{subgrup} al lui $G$ (notat $H \le G$) dacă:
\begin{enumerate}
    \item $(\forall) x, y \in H \implies x \cdot y \in H$ ($H$ este parte stabilă în raport cu $\cdot$).
    \item $(\forall) x \in H \implies x^{-1} \in H$.
\end{enumerate}
\end{definition}

\begin{remark}
Evident: $e \in H$. $(x \in H, x^{-1} \in H \implies x \cdot x^{-1} \in H \implies e \in H)$.
\end{remark}

\begin{theorem}[de caracterizare a subgrupurilor]
$$ H \le G \iff (\forall) x, y \in H \implies x \cdot y^{-1} \in H $$
\end{theorem}

\begin{proof}
``$\implies$'': $y \in H \implies y^{-1} \in H \implies x \cdot y^{-1} \in H$. \\
``$\impliedby$'': Presupunem că $x \cdot y^{-1} \in H$, $(\forall) x, y \in H$.
\begin{itemize}
    \item $\implies$ (pentru $x=y$) $x \cdot x^{-1} = e \in H$.
    \item $e \in H, x \in H \implies e \cdot x^{-1} = x^{-1} \in H$ (2).
    \item din $x, y \in H \implies x \in H$ și $y^{-1} \in H \implies x \cdot (y^{-1})^{-1} \in H \implies x \cdot y \in H$ (1).
\end{itemize}
$\implies H \le G$.
\end{proof}

\begin{example}\leavevmode
\begin{itemize}
    \item[(1)] $(\mathbb{Z}, +) \le (\mathbb{Q}, +) \le (\mathbb{R}, +) \le (\mathbb{C}, +)$.
    \item[(2)] $n \in \mathbb{N}, n\mathbb{Z} = \{nk \mid k \in \mathbb{Z}\} \le (\mathbb{Z}, +)$. \\
    \textit{Demonstrație:} $x, y \in n\mathbb{Z} \implies x = nk_1, y = nk_2 \implies x - y = n(k_1 - k_2) \in n\mathbb{Z}$.
    \item[(3)] $U_n = \{z \in \mathbb{C} \mid z^n = 1\} \le (\mathbb{C}^*, \cdot)$.
\end{itemize}
\end{example}

\subsection*{Morfisme de grupuri. Izomorfisme}

\begin{definition}
Fie $(G, \cdot)$ și $(G', *)$ două grupuri. O aplicație $f: G \rightarrow G'$ se numește \textbf{morfism de grupuri} dacă:
$$ f(x \cdot y) = f(x) * f(y), \quad (\forall) x, y \in G $$
\end{definition}

\begin{proposition}
Dacă $f: G \rightarrow G'$ este morfism $\implies$
\begin{enumerate}
    \item $f(e) = e'$
    \item $f(x^{-1}) = (f(x))^{-1}, \quad (\forall) x \in G$
\end{enumerate}
\end{proposition}

\begin{proof}
1) $f(x) = f(x \cdot e) = f(x) * f(e) \implies e' = f(e)$. \\
2) $e' = f(e) = f(x \cdot x^{-1}) = f(x) * f(x^{-1}) \implies f(x^{-1}) = (f(x))^{-1}$.
\end{proof}

\begin{definition}
Un morfism bijectiv se numește \textbf{izomorfism}. ($G \cong G'$).
\end{definition}

\begin{example}\leavevmode
\begin{itemize}
    \item[(1)] $f: \mathbb{Z} \rightarrow 2\mathbb{Z}, f(x) = 2x$. \\
    \textit{Demonstrație:} $f(x + y) = 2(x + y) = 2x + 2y = f(x) + f(y) \implies$ morfism; bijectivă $\implies$ izomorfism.
    \item[(2)] $f: (\mathbb{R}, +) \rightarrow ((0, \infty), \cdot), f(x) = e^x$. \\
    \textit{Demonstrație:} $f(x + y) = e^{x+y} = e^x \cdot e^y = f(x) \cdot f(y) \implies$ morfism; bijectivă $\implies$ izomorfism.
    \item[(3)] $f: ((0, \infty), \cdot) \rightarrow (\mathbb{R}, +), f(x) = \ln x$. \\
    \textit{Demonstrație:} $f(x \cdot y) = \ln(x \cdot y) = \ln x + \ln y = f(x) + f(y) \implies$ morfism; bijectivă $\implies$ izomorfism.
\end{itemize}
\end{example}
